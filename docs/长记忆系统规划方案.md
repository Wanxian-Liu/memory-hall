```python
# introspection/module_map.py
class ModuleMapGenerator:
    """自动扫描并生成Mimir-Core模块地图"""
    
    def __init__(self, root_path: Path):
        self.root_path = root_path
        self.modules: Dict[str, ModuleInfo] = {}
    
    def scan(self) -> Dict[str, ModuleInfo]:
        """扫描所有Python模块"""
        for py_file in self.root_path.rglob("*.py"):
            # 跳过__pycache__、.pytest_cache等
            if self._should_skip(py_file):
                continue
            
            module_id = self._compute_module_id(py_file)
            module_info = self._analyze_module(py_file)
            self.modules[module_id] = module_info
        
        return self.modules
    
    def _analyze_module(self, py_file: Path) -> ModuleInfo:
        """分析单个模块"""
        tree = ast.parse(py_file.read_text())
        classes = [n.name for n in ast.walk(tree) if isinstance(n, ast.ClassDef)]
        functions = [n.name for n in ast.walk(tree) if isinstance(n, ast.FunctionDef)]
        imports = self._extract_imports(tree)
        
        return ModuleInfo(
            module_id=self._compute_module_id(py_file),
            name=py_file.stem,
            path=py_file,
            classes=classes,
            functions=functions,
            interfaces=self._detect_interfaces(classes),
            status=ModuleStatus.UNKNOWN,
            health_score=1.0,
            last_updated=datetime.now()
        )
    
    def build_dependency_graph(self) -> Dict[str, List[str]]:
        """构建模块依赖图"""
        graph = defaultdict(list)
        for module_id, info in self.modules.items():
            for dep in info.dependencies:
                graph[module_id].append(dep)
        return dict(graph)
```

```python
# introspection/status.py
from enum import Enum
from dataclasses import dataclass
from typing import Callable, Dict, List
import asyncio

class ModuleStatus(Enum):
    ACTIVE = "active"       # 运行正常
    SLEEP = "sleep"         # 空闲/待激活
    ERROR = "error"         # 有错误
    UNKNOWN = "unknown"     # 未知
    DEGRADED = "degraded"   # 降级运行

@dataclass
class HealthProbeResult:
    module_id: str
    status: ModuleStatus
    health_score: float      # 0.0-1.0
    latency_ms: float
    error_message: Optional[str]
    timestamp: datetime

def status_probe(module_id: str):
    """装饰器：标记模块健康探针"""
    def decorator(func):
        func._is_status_probe = True
        func._module_id = module_id
        return func
    return decorator

class StatusMonitor:
    """状态监控器 - 借鉴claw-code的health check"""
    
    def __init__(self):
        self.probes: Dict[str, Callable] = {}
        self.subscribers: List[Callable] = []
        self._cache: Dict[str, HealthProbeResult] = {}
        self._cache_ttl = 30  # 30秒缓存
    
    def register_probe(self, module_id: str, probe_fn: Callable):
        """注册健康探针"""
        self.probes[module_id] = probe_fn
    
    async def probe_module(self, module_id: str) -> HealthProbeResult:
        """执行单个模块探针"""
        if module_id in self._cache:
            cached = self._cache[module_id]
            if (datetime.now() - cached.timestamp).seconds < self._cache_ttl:
                return cached
        
        probe = self.probes.get(module_id)
        if not probe:
            return HealthProbeResult(
                module_id=module_id,
                status=ModuleStatus.UNKNOWN,
                health_score=0.0,
                latency_ms=0,
                error_message="No probe registered",
                timestamp=datetime.now()
            )
        
        start = time.time()
        try:
            result = await probe()
            latency = (time.time() - start) * 1000
            
            health_result = HealthProbeResult(
                module_id=module_id,
                status=ModuleStatus.ACTIVE if result else ModuleStatus.ERROR,
                health_score=1.0 if result else 0.0,
                latency_ms=latency,
                error_message=None,
                timestamp=datetime.now()
            )
        except Exception as e:
            health_result = HealthProbeResult(
                module_id=module_id,
                status=ModuleStatus.ERROR,
                health_score=0.0,
                latency_ms=(time.time() - start) * 1000,
                error_message=str(e),
                timestamp=datetime.now()
            )
        
        self._cache[module_id] = health_result
        return health_result
    
    async def probe_all(self) -> Dict[str, HealthProbeResult]:
        """并行执行所有探针"""
        tasks = [self.probe_module(mid) for mid in self.probes]
        results = await asyncio.gather(*tasks)
        return {r.module_id: r for r in results}
    
    def subscribe(self, callback: Callable[[str, ModuleStatus], None]):
        """订阅状态变化通知"""
        self.subscribers.append(callback)
```

```python
# introspection/problem_detector.py
from enum import Enum
from dataclasses import dataclass
from typing import List, Dict
import re

class ProblemSeverity(Enum):
    CRITICAL = "critical"   # 立即处理
    HIGH = "high"           # 24小时内
    MEDIUM = "medium"       # 本周内
    LOW = "low"             # 稍后处理

class ProblemClass(Enum):
    """借鉴claw-code的LaneFailureClass"""
    IMPORT_ERROR = "import_error"
    SYNTAX_ERROR = "syntax_error"
    RUNTIME_ERROR = "runtime_error"
    TIMEOUT = "timeout"
    MEMORY_LEAK = "memory_leak"
    DEPENDENCY_MISSING = "dependency_missing"
    API_MISMATCH = "api_mismatch"
    PERMISSION_DENIED = "permission_denied"
    UNKNOWN = "unknown"

# 错误模式库 - 借鉴claw-code的RecoveryRecipes
ERROR_PATTERNS = [
    {
        "class": ProblemClass.IMPORT_ERROR,
        "pattern": r"ImportError: cannot import name '(\w+)'",
        "suggestion": "检查模块是否安装或导入路径是否正确",
        "severity": ProblemSeverity.HIGH
    },
    {
        "class": ProblemClass.SYNTAX_ERROR,
        "pattern": r"SyntaxError: (.+)",
        "suggestion": "修复语法错误",
        "severity": ProblemSeverity.CRITICAL
    },
    {
        "class": ProblemClass.TIMEOUT,
        "pattern": r"timeout|Timeout|timed out",
        "suggestion": "增加超时时间或优化执行逻辑",
        "severity": ProblemSeverity.MEDIUM
    },
    {
        "class": ProblemClass.MEMORY_LEAK,
        "pattern": r"MemoryError|memory leak",
        "suggestion": "检查内存泄漏，增加内存限制",
        "severity": ProblemSeverity.HIGH
    },
]

@dataclass
class ProblemReport:
    problem_id: str
    severity: ProblemSeverity
    problem_class: ProblemClass
    module_id: str
    description: str
    symptoms: List[str]
    suggested_fix: str
    confidence: float        # 0.0-1.0
    raw_error: str           # 原始错误信息
    timestamp: datetime

class ProblemDetector:
    """问题检测器 - 借鉴claw-code的error classification"""
    
    def __init__(self, log_path: Path):
        self.log_path = log_path
        self.problem_history: List[ProblemReport] = []
    
    def detect_from_log(self, log_content: str) -> List[ProblemReport]:
        """从日志中检测问题"""
        problems = []
        for line in log_content.split('\n'):
            for pattern_def in ERROR_PATTERNS:
                match = re.search(pattern_def["pattern"], line, re.IGNORECASE)
                if match:
                    problem = self._create_problem(pattern_def, line)
                    problems.append(problem)
        return problems
    
    def classify(self, error_log: str) -> ProblemClass:
        """分类错误类型 - 借鉴claw-code的LaneFailureClass"""
        for pattern_def in ERROR_PATTERNS:
            if re.search(pattern_def["pattern"], error_log, re.IGNORECASE):
                return ProblemClass(pattern_def["class"])
        return ProblemClass.UNKNOWN
    
    def root_cause(self, problem: ProblemReport) -> Dict:
        """根因分析 - 返回修复链路"""
        return {
            "problem": problem.problem_id,
            "cause": self._analyze_cause(problem),
            "fix_steps": self._generate_fix_steps(problem),
            "prevention": self._suggest_prevention(problem)
        }
```

```python
class FIXEngine:
    """自动修复损坏/过时的Capsule"""
    
    async def fix(self, capsule_path: str, problem: ProblemReport) -> FixResult:
        # Phase 1: 定位问题
        error_class = problem.problem_class
        error_context = await self._extract_error_context(capsule_path, problem)
        
        # Phase 2: 生成修复指令（借鉴OpenSpace的fix_generator）
        fix_template = self._get_fix_template(error_class)
        fix_instruction = await self._generate_fix(
            error_class, error_context, fix_template
        )
        
        # Phase 3: 应用修复（带重试）
        for attempt in range(self.max_retries):
            try:
                fixed_content = await self._apply_fix(
                    capsule_path, fix_instruction
                )
                
                # Phase 4: 验证修复
                if await self._verify_fix(capsule_path, fixed_content):
                    await self._save_fixed_version(capsule_path, fixed_content)
                    return FixResult(
                        success=True,
                        original_path=capsule_path,
                        fixed_path=self._get_fixed_path(capsule_path),
                        steps_taken=[f"attempt_{attempt+1}"],
                        verification_passed=True
                    )
            except Exception as e:
                if attempt == self.max_retries - 1:
                    return FixResult(
                        success=False,
                        original_path=capsule_path,
                        fixed_path=None,
                        steps_taken=[],
                        verification_passed=False
                    )
        
        return FixResult(success=False, original_path=capsule_path)
```

#### 1.4.4 FIX引擎详细设计

```python
# evolve/fix_engine.py
from pathlib import Path
from typing import Optional
import asyncio

class FIXEngine:
    """FIX自动修复引擎 - 借鉴OpenSpace + claw-code"""
    
    MAX_RETRIES = 3
    BACKOFF_SECONDS = [1, 2, 5]  # 指数退避
    
    def __init__(self, capsule_dir: Path, backup_dir: Path):
        self.capsule_dir = capsule_dir
        self.backup_dir = backup_dir
        self.fix_templates = self._load_fix_templates()
    
    async def fix(self, capsule_path: str, problem: ProblemReport) -> FixResult:
        """主修复流程"""
        # 1. 创建备份
        await self._create_backup(capsule_path)
        
        # 2. 定位问题
        error_context = await self._extract_context(capsule_path, problem)
        
        # 3. 获取修复模板
        template = self._get_template(problem.problem_class)
        
        # 4. 生成修复
        fix_content = await self._generate_fix_content(
            capsule_path, error_context, template
        )
        
        # 5. 验证并保存
        if await self._verify(capsule_path, fix_content):
            await self._save(capsule_path, fix_content)
            return FixResult(success=True, verification_passed=True)
        
        return FixResult(success=False, verification_passed=False)
    
    async def _apply_fix_with_retry(self, capsule_path: str, instruction: str) -> str:
        """带重试的修复应用 - 借鉴claw-code的retry机制"""
        for attempt in range(self.MAX_RETRIES):
            try:
                result = await self._apply_single_fix(capsule_path, instruction)
                if await self._verify(capsule_path, result):
                    return result
            except Exception as e:
                wait_time = self.BACKOFF_SECONDS[attempt]
                await asyncio.sleep(wait_time)
```

```python
# agent/worker_manager.py
from enum import Enum
from dataclasses import dataclass
from typing import Optional, Dict, List
import asyncio
import subprocess
import time

class WorkerState(Enum):
    """借鉴claw-code的Worker状态机"""
    SPAWNING = "spawning"           # 创建中
    TRUST_REQUIRED = "trust_required" # 等待信任门
    READY = "ready"                 # 就绪
    RUNNING = "running"             # 执行中
    RESTART = "restart"            # 重启中
    COMPLETE = "complete"           # 完成
    FAILED = "failed"              # 失败

@dataclass
class WorkerConfig:
    worker_id: str
    role: str                       # 来自178角色库
    agent_id: str                   # openclaw agent id
    max_retries: int = 3
    timeout_seconds: int = 300
    workspace: Optional[Path] = None

@dataclass
class Worker:
    worker_id: str
    config: WorkerConfig
    state: WorkerState
    process: Optional[subprocess.Popen] = None
    started_at: Optional[datetime] = None
    last_heartbeat: Optional[datetime] = None
    restart_count: int = 0

class WorkerManager:
    """Worker生命周期管理器 - 完全借鉴claw-code的worker_boot.rs"""
    
    def __init__(self, workspace_root: Path):
        self.workspace_root = workspace_root
        self.workers: Dict[str, Worker] = {}
        self.trust_gate_handler = TrustGateHandler()
    
    async def spawn(self, role: str, config: WorkerConfig) -> Worker:
        """启动Worker - 借鉴claw-code的spawn流程"""
        # 1. 创建Worker记录
        worker = Worker(
            worker_id=config.worker_id,
            config=config,
            state=WorkerState.SPAWNING
        )
        self.workers[config.worker_id] = worker
        
        # 2. 启动子进程
        process = await self._start_process(config)
        worker.process = process
        worker.started_at = datetime.now()
        worker.state = WorkerState.SPAWNING
        
        # 3. 等待TrustRequired状态
        await self._wait_for_trust_required(worker)
        
        # 4. 检测信任门
        trust_result = await self.trust_gate_handler.detect(process)
        if trust_result.needs_manual_approval:
            worker.state = WorkerState.TRUST_REQUIRED
            return worker
        
        # 5. 自动解决信任门
        await self.trust_gate_handler.auto_approve(process)
        worker.state = WorkerState.READY
        
        return worker
    
    async def send(self, worker_id: str, task: Task) -> None:
        """发送任务到Worker"""
        worker = self.workers.get(worker_id)
        if not worker or worker.state != WorkerState.READY:
            raise WorkerNotReadyError(worker_id)
        
        worker.state = WorkerState.RUNNING
        await self._send_task_to_process(worker.process, task)
    
    async def observe(self, worker_id: str) -> WorkerState:
        """观察Worker状态 - 借鉴claw-code的observe机制"""
        worker = self.workers.get(worker_id)
        if not worker:
            return WorkerState.FAILED
        
        if worker.process.poll() is not None:
            # 进程已结束
            worker.state = WorkerState.COMPLETE
            return worker.state
        
        # 检测运行中信号
        output = await self._read_process_output(worker.process)
        
        if "Ready for prompt" in output or "ReadyForPrompt" in output:
            worker.state = WorkerState.READY
        elif "Working" in output or "Thinking" in output:
            worker.state = WorkerState.RUNNING
        elif "error" in output.lower() or "failed" in output.lower():
            worker.state = WorkerState.FAILED
        
        worker.last_heartbeat = datetime.now()
        return worker.state
    
    async def restart(self, worker_id: str) -> bool:
        """重启Worker - 借鉴claw-code的restart机制"""
        worker = self.workers.get(worker_id)
        if not worker:
            return False
        
        if worker.restart_count >= worker.config.max_retries:
            worker.state = WorkerState.FAILED
            return False
        
        # 终止旧进程
        await self._terminate_process(worker.process)
        
        # 重启
        worker.restart_count += 1
        worker.state = WorkerState.RESTART
        
        new_worker = await self.spawn(
            role=worker.config.role,
            config=worker.config
        )
        
        return new_worker.state == WorkerState.READY
    
    async def terminate(self, worker_id: str) -> None:
        """终止Worker"""
        worker = self.workers.get(worker_id)
        if worker and worker.process:
            await self._terminate_process(worker.process)
            worker.state = WorkerState.FAILED
```

```python
from dataclasses import dataclass
from typing import List, Optional
import asyncio

@dataclass
class Task:
    task_id: str
    description: str
    priority: int              # 1-10, 10最高
    worker_role: str           # 需要的角色
    estimated_duration: int     # 秒
    dependencies: List[str]    # 前置任务ID

class TaskScheduler:
    """智能任务调度器"""
    
    def __init__(self, worker_manager: WorkerManager):
        self.worker_manager = worker_manager
        self.task_queue: List[Task] = []
        self.running_tasks: Dict[str, Task] = {}
        self.completed_tasks: Dict[str, Task] = {}
    
    async def submit(self, task: Task) -> None:
        """提交任务"""
        self.task_queue.append(task)
        self.task_queue.sort(key=lambda t: t.priority, reverse=True)
    
    async def schedule(self) -> Optional[Task]:
        """调度下一个任务 - 借鉴claw-code的智能调度"""
        # 1. 找到就绪的任务（依赖已满足）
        for task in self.task_queue:
            if self._are_dependencies_met(task):
                # 2. 找到合适的Worker
                worker = await self._find_available_worker(task.worker_role)
                
                if worker:
                    # 3. 分配任务
                    self.task_queue.remove(task)
                    self.running_tasks[task.task_id] = task
                    
                    await self.worker_manager.send(worker.worker_id, task)
                    return task
        
        return None
    
    async def _find_available_worker(self, role: str) -> Optional[Worker]:
        """找到可用的Worker - 负载均衡"""
        available = [
            w for w in self.worker_manager.workers.values()
            if w.state == WorkerState.READY and w.config.role == role
        ]
        
        if not available:
            return None
        
        # 选择负载最低的（重启次数最少）
        return min(available, key=lambda w: w.restart_count)
    
    def _are_dependencies_met(self, task: Task) -> bool:
        """检查依赖是否满足"""
        for dep_id in task.dependencies:
            if dep_id not in self.completed_tasks:
                return False
        return True
```

```python
from enum import Enum
from dataclasses import dataclass
from typing import List, Set, Optional
import importlib
import time

class BootstrapPhase(Enum):
    """借鉴claw-code的BootstrapPhase"""
    CORE = "core"                 # 核心模块（必须）
    MEMORY = "memory"             # 记忆模块
    INTROSPECTION = "introspection"  # 自省模块
    SENSORY = "sensory"           # 感知模块
    EXTRACTOR = "extractor"       # 萃取模块
    EVOLVE = "evolve"             # 进化模块

class BootstrapMode(Enum):
    MINIMAL = "minimal"   # 只加载核心+记忆+自省
    STANDARD = "standard" # 标准模式（不含Worker）
    FULL = "full"         # 全量加载

@dataclass
class BootstrapResult:
    mode: BootstrapMode
    phases_loaded: List[BootstrapPhase]
    total_time_ms: float
    modules_loaded: int
    memory_mb: float
    success: bool
    error: Optional[str] = None

class BootstrapFastPath:
    """增量加载引导 - 完全借鉴claw-code的bootstrap.rs"""
    
    # Phase依赖关系
    PHASE_DEPS = {
        BootstrapPhase.CORE: set(),
        BootstrapPhase.MEMORY: {BootstrapPhase.CORE},
        BootstrapPhase.INTROSPECTION: {BootstrapPhase.CORE},
        BootstrapPhase.SENSORY: {BootstrapPhase.CORE, BootstrapPhase.MEMORY},
        BootstrapPhase.EXTRACTOR: {BootstrapPhase.CORE},
        BootstrapPhase.EVOLVE: {BootstrapPhase.CORE, BootstrapPhase.INTROSPECTION},
        BootstrapPhase.HEALTH: {BootstrapPhase.INTROSPECTION},
        BootstrapPhase.WORKER: {BootstrapPhase.CORE, BootstrapPhase.MEMORY},
        BootstrapPhase.MAIN: {BootstrapPhase.CORE},
    }
    
    # Mode对应的Phase集合
    MODE_PHASES = {
        BootstrapMode.MINIMAL: {
            BootstrapPhase.CORE,
            BootstrapPhase.MEMORY,
            BootstrapPhase.INTROSPECTION,
        },
        BootstrapMode.STANDARD: {
            BootstrapPhase.CORE,
            BootstrapPhase.MEMORY,
            BootstrapPhase.INTROSPECTION,
            BootstrapPhase.SENSORY,
            BootstrapPhase.EXTRACTOR,
            BootstrapPhase.EVOLVE,
            BootstrapPhase.HEALTH,
        },
        BootstrapMode.FULL: set(BootstrapPhase),
    }
    
    def __init__(self, project_root: Path):
        self.project_root = project_root
        self.loaded_modules: Set[str] = set()
        self.phase_timings: Dict[BootstrapPhase, float] = {}
    
    def boot(self, mode: BootstrapMode = BootstrapMode.MINIMAL) -> BootstrapResult:
        """主引导流程"""
        start_time = time.time()
        phases_to_load = self._resolve_phases(mode)
        phases_loaded = []
        
        try:
            for phase in phases_to_load:
                phase_start = time.time()
                self._load_phase(phase)
                self.phase_timings[phase] = (time.time() - phase_start) * 1000
                phases_loaded.append(phase)
            
            total_time = (time.time() - start_time) * 1000
            
            return BootstrapResult(
                mode=mode,
                phases_loaded=phases_loaded,
                total_time_ms=total_time,
                modules_loaded=len(self.loaded_modules),
                memory_mb=self._get_memory_usage(),
                success=True
            )
        except Exception as e:
            return BootstrapResult(
                mode=mode,
                phases_loaded=phases_loaded,
                total_time_ms=(time.time() - start_time) * 1000,
                modules_loaded=len(self.loaded_modules),
                memory_mb=0,
                success=False,
                error=str(e)
            )
    
    def _resolve_phases(self, mode: BootstrapMode) -> List[BootstrapPhase]:
        """解析需要加载的Phase（去重+保序）"""
        target_phases = self.MODE_PHASES[mode]
        resolved = []
        seen: Set[BootstrapPhase] = set()
        
        def add_with_deps(phase: BootstrapPhase):
            for dep in self.PHASE_DEPS[phase]:
                if dep not in seen:
                    add_with_deps(dep)
            if phase not in seen:
                seen.add(phase)
                resolved.append(phase)
        
        for phase in target_phases:
            add_with_deps(phase)
        
        return resolved
    
    def _load_phase(self, phase: BootstrapPhase) -> None:
        """加载单个Phase"""
        modules = self.PHASE_MODULES[phase]
        for module_path in modules:
            self._lazy_import(module_path)
    
    def _lazy_import(self, module_path: str) -> None:
        """惰性导入模块"""
        if module_path in self.loaded_modules:
            return
        
        importlib.import_module(module_path)
        self.loaded_modules.add(module_path)
    
    # Phase到模块的映射
    PHASE_MODULES = {
        BootstrapPhase.CORE: [
            "mimir_core.config",
            "mimir_core.base_wal",
            "mimir_core.utils",
        ],
        BootstrapPhase.MEMORY: [
            "mimir_core.memory_layer",

            "mimir_core.extractor",
        ],
        BootstrapPhase.INTROSPECTION: [
            "mimir_core.introspection.modules",
```
